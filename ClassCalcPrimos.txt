public CalculoPrimos() {
}
	public static void main(String a[]){
		int n=1, n2, cont;
		float r=0;
		while(n < 100){
			n2=n-1;
			for(cont = n2; cont >= 2; cont–) {
				r = n % cont;
				if(r == 0) {
					System.out.println("O numero "+ n +" nao e primo!");
					cont=1;
				}
			}
			if(r!=0) {
				System.out.println("O numero "+ n +" e primo!");
			}
			n++;
		}
	}
}
